## 5. 집합 연산자
### 5. 집합 연산자
- **UNION**: 합집합(중복 제거)
- **UNION ALL**: 합집합(중복 포함)
- **INTERSECT**: 교집합
- **MINUS/EXCEPT**: 차집합
```
-- 샘플 테이블 생성
CREATE TABLE employees (id NUMBER, name VARCHAR2(20), salary NUMBER);
CREATE TABLE managers (id NUMBER, name VARCHAR2(20), salary NUMBER);

INSERT INTO employees VALUES (1, '김철수', 3000);
INSERT INTO employees VALUES (2, '이영희', 4000);
INSERT INTO employees VALUES (3, '박영수', 5000);
INSERT INTO employees VALUES (4, '최유진', 3500);
INSERT INTO employees VALUES (5, '정민호', 4500);

INSERT INTO managers VALUES (3, '박영수', 5000);
INSERT INTO managers VALUES (4, '최유진', 3500);
INSERT INTO managers VALUES (6, '김미영', 6000);

-- UNION (중복 제거)
SELECT id, name FROM employees WHERE salary > 3000
UNION
SELECT id, name FROM managers WHERE salary > 3000;
/*
결과:
3 박영수
4 최유진
5 정민호
6 김미영
*/

-- UNION ALL (중복 포함)
SELECT id, name FROM employees WHERE salary > 3000
UNION ALL
SELECT id, name FROM managers WHERE salary > 3000;
/*
결과:
2 이영희
3 박영수
4 최유진
5 정민호
3 박영수
4 최유진
6 김미영
*/

-- INTERSECT (교집합)
SELECT id FROM employees
INTERSECT
SELECT id FROM managers;
/*
결과:
3
4
*/

-- MINUS/EXCEPT (차집합) 
-- Oracle: MINUS / SQL Server: EXCEPT
SELECT id FROM employees
MINUS
SELECT id FROM managers;
/*
결과:
1
2
5
*/
```
